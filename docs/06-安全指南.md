## 安全概述

### 1. 安全架构

UniAdmin 采用多层次安全架构，确保系统在各个层面都得到保护：

```
┌─────────────────────────────────────────┐
│         应用层安全                       │
│  - 输入验证                              │
│  - 输出编码                              │
│  - 业务逻辑验证                          │
└─────────────────────────────────────────┘
                    ↓
┌─────────────────────────────────────────┐
│         授权层安全                        │
│  - RBAC 权限检查                         │
│  - 资源访问控制                          │
│  - 操作权限验证                          │
└─────────────────────────────────────────┘
                    ↓
┌─────────────────────────────────────────┐
│         认证层安全                        │
│  - 用户身份验证                          │
│  - 会话管理                              │
│  - Token 验证                            │
└─────────────────────────────────────────┘
                    ↓
┌─────────────────────────────────────────┐
│         数据层安全                        │
│  - SQL 注入防护                          │
│  - 参数化查询                            │
│  - 数据加密                              │
└─────────────────────────────────────────┘
                    ↓
┌─────────────────────────────────────────┐
│         网络层安全                        │
│  - HTTPS/TLS                            │
│  - 防火墙规则                           │
│  - DDoS 防护                            │
└─────────────────────────────────────────┘
```

### 2. 安全目标

- **机密性**: 保护敏感数据不被未授权访问
- **完整性**: 确保数据不被篡改
- **可用性**: 确保系统持续可用
- **可追溯性**: 记录所有操作以便审计
- **合规性**: 符合相关安全标准和法规

### 3. 安全合规

UniAdmin 遵循以下安全标准和最佳实践：

- OWASP Top 10
- ISO 27001 信息安全管理体系
- GDPR 数据保护条例
- 等保 2.0（中国网络安全等级保护）

#### 密码验证实现
```delphi
function TPasswordValidator.Validate(const Password: string): Boolean;
var
  HasUpper, HasLower, HasNumber, HasSpecial: Boolean;
  I: Integer;
  Char: Char;
begin
  Result := False;

  // 检查长度
  if Length(Password) < FMinLength then
    raise EPasswordException.Create('密码长度不能少于 ' + IntToStr(FMinLength) + ' 位');

  if Length(Password) > FMaxLength then
    raise EPasswordException.Create('密码长度不能超过 ' + IntToStr(FMaxLength) + ' 位');

  // 检查字符类型
  HasUpper := False;
  HasLower := False;
  HasNumber := False;
  HasSpecial := False;

  for I := 1 to Length(Password) do
  begin
    Char := Password[I];
    if Char in ['A'..'Z'] then
      HasUpper := True
    else if Char in ['a'..'z'] then
      HasLower := True
    else if Char in ['0'..'9'] then
      HasNumber := True
    else if Char in ['!', '@', '#', '$', '%', '^', '&', '*', '(', ')', '-', '_', '+', '='] then
      HasSpecial := True;
  end;

  if FRequireUppercase and not HasUpper then
    raise EPasswordException.Create('密码必须包含大写字母');

  if FRequireLowercase and not HasLower then
    raise EPasswordException.Create('密码必须包含小写字母');

  if FRequireNumbers and not HasNumber then
    raise EPasswordException.Create('密码必须包含数字');

  if FRequireSpecialChars and not HasSpecial then
    raise EPasswordException.Create('密码必须包含特殊字符');

  // 检查是否在禁止列表中
  if IsPasswordForbidden(Password) then
    raise EPasswordException.Create('该密码过于简单，请使用其他密码');

  Result := True;
end;
```

### 3. 登录安全

#### 登录失败限制
```delphi
procedure TAuthenticationService.HandleLogin(const Username, Password: string;
  const IPAddress: string);
var
  User: TUser;
  FailedAttempts: Integer;
begin
  // 检查登录失败次数
  FailedAttempts := GetFailedLoginAttempts(Username, IPAddress);
  if FailedAttempts >= MaxFailedAttempts then
  begin
    LogSecurityEvent('登录被锁定', Username, IPAddress);
    raise ELoginException.Create('登录失败次数过多，账户已被锁定');
  end;

  // 验证用户
  User := GetUserByUsername(Username);
  if User = nil then
  begin
    IncrementFailedLoginAttempts(Username, IPAddress);
    raise ELoginException.Create('用户名或密码错误');
  end;

  // 验证密码
  if not PasswordHasher.VerifyPassword(Password, User.PasswordHash) then
  begin
    IncrementFailedLoginAttempts(Username, IPAddress);
    LogSecurityEvent('登录失败', Username, IPAddress);
    raise ELoginException.Create('用户名或密码错误');
  end;

  // 检查账户状态
  if not User.IsActive then
    raise ELoginException.Create('账户已被禁用');

  // 清除失败记录
  ClearFailedLoginAttempts(Username, IPAddress);

  // 创建会话
  CreateSession(User);
end;
```

```delphi
function TAuthorizationService.CheckPermission(UserId: Integer;
  const PermissionCode: string): Boolean;
var
  UserPermissions: TList<string>;
begin
  // 获取用户权限
  UserPermissions := GetUserPermissions(UserId);

  // 检查权限
  Result := UserPermissions.Contains(PermissionCode);

  // 记录权限检查

function TAuthorizationService.CheckPermission(UserId: Integer;
  const PermissionCodes: array of string): Boolean;
var
  I: Integer;
begin
  Result := True;
  for I := Low(PermissionCodes) to High(PermissionCodes) do
  begin
    if not CheckPermission(UserId, PermissionCodes[I]) then
    begin
      Result := False;
      Break;
    end;
  end;
end;
```

### 3. 数据权限

#### 行级安全
```sql
-- 创建行级安全策略
CREATE SCHEMA Security;

CREATE FUNCTION Security.UserHasAccess(@UserId INT, @TargetUserId INT)
RETURNS BIT
AS
BEGIN
    DECLARE @HasAccess BIT = 0;

    -- 系统管理员可以访问所有数据
    IF EXISTS (
        SELECT 1 FROM UserRoles ur
        JOIN Roles r ON ur.RoleId = r.Id
        WHERE ur.UserId = @UserId AND r.Name = 'admin'
    )
    SET @HasAccess = 1;

    -- 用户只能访问自己的数据
    IF @UserId = @TargetUserId
    SET @HasAccess = 1;

    RETURN @HasAccess;
END;
```

#### 数据过滤

```delphi
var
  RequiredPermissions: TArray<string>;
  UserPermissions: TList<string>;
  I: Integer;
begin
  // 获取所需权限
  RequiredPermissions := GetRequiredPermissions(Context);
```

### 4. DDoS 防护

#### 速率限制
```delphi
type
  TRateLimiter = class
  private
    FRequests: TDictionary<string, TQueue<DateTime>>;
    FMaxRequests: Integer;
    FTimeWindow: Integer;
  public
    constructor Create(MaxRequests, TimeWindow: Integer);
    function IsAllowed(const IPAddress: string): Boolean;
  end;

function TRateLimiter.IsAllowed(const IPAddress: string): Boolean;
var
  Requests: TQueue<DateTime>;
  Now: DateTime;
begin
  Now := Now;

  if not FRequests.TryGetValue(IPAddress, Requests) then
  begin
    Requests := TQueue<DateTime>.Create;
    FRequests.Add(IPAddress, Requests);
  end;

  // 清理过期请求
  while (Requests.Count > 0) and (MillisecondsBetween(Now, Requests.Peek) > FTimeWindow) do
    Requests.Dequeue;

  // 检查是否超过限制
  if Requests.Count >= FMaxRequests then
    Result := False
  else
  begin
    Requests.Enqueue(Now);
    Result := True;
  end;
end;
```

```delphi
function TInputValidator.Validate(const Input: string; const Rules: TValidationRules): Boolean;
begin
  Result := True;

  if (Rules.MinLength > 0) and (Length(Input) < Rules.MinLength) then
    raise EValidationException.Create('长度不能少于 ' + IntToStr(Rules.MinLength) + ' 位');

  if (Rules.MaxLength > 0) and (Length(Input) > Rules.MaxLength) then
    raise EValidationException.Create('长度不能超过 ' + IntToStr(Rules.MaxLength) + ' 位');

  if (Rules.Pattern <> '') and not TRegex.IsMatch(Input, Rules.Pattern) then
    raise EValidationException.Create('格式不正确');

  // 检查自定义验证器
  if Assigned(Rules.CustomValidator) and not Rules.CustomValidator(Input) then
    raise EValidationException.Create('验证失败');
end;
```
